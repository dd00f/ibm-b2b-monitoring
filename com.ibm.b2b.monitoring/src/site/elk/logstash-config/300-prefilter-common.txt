
filter {

	if (![@metadata][local_timezone]) {
		mutate {
			add_field => { "[@metadata][local_timezone]" => "America/New_York" }
		}
	}
	
	if ([host]) {
		grok {
			match => { "host" => "(?<shortHost>[^\.]{1,})\.{0,}.*" }
		}
	}

	metrics {
		meter => "totalEvents-%{[@metadata][beat]}"
		add_tag => "logstash-metric"
		flush_interval => 60
		clear_interval => 60
	}
    metrics {
		meter => "totalEvents"
		add_tag => "logstash-metric"
		flush_interval => 60
		clear_interval => 60
	}

}
